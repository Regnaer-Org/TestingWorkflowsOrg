LastTeamAsOfDay =
VAR CurrentIssue = MetaRisk_GitHub_Data[issue_id]
VAR CurrentDay = SELECTEDVALUE('Calendar'[Date])
RETURN
CALCULATE(
    LASTNONBLANK(issue_changelog[change_date], 1),
    issue_changelog[issue_id] = CurrentIssue,
    issue_changelog[field_changed] = "team",
    issue_changelog[change_date] <= CurrentDay
)
