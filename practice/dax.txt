Sprint Completion (All In Sprint) =
VAR SelectedSprint = SELECTEDVALUE(SprintmetaData[sprint])
VAR SprintStart = CALCULATE(MIN(SprintmetaData[startdate]), SprintmetaData[sprint] = SelectedSprint)
VAR SprintEnd = CALCULATE(MAX(SprintmetaData[enddate]), SprintmetaData[sprint] = SelectedSprint)
RETURN
CALCULATE(
    COUNTROWS(MetaRisk_GitHub_Data),
    FILTER(
        MetaRisk_GitHub_Data,
        MetaRisk_GitHub_Data[sprint] = SelectedSprint &&
        NOT ISBLANK(MetaRisk_GitHub_Data[content_closed_at]) &&
        MetaRisk_GitHub_Data[content_closed_at] >= SprintStart &&
        MetaRisk_GitHub_Data[content_closed_at] <= SprintEnd
    )
)
