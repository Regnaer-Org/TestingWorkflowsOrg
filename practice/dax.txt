InSprintDateRange =
VAR SelectedSprint = SELECTEDVALUE(SprintmetaData[sprint])
VAR SprintStart =
    CALCULATE(
        MIN(SprintmetaData[start_date]),
        SprintmetaData[sprint] = SelectedSprint
    )
VAR SprintEnd =
    CALCULATE(
        MAX(SprintmetaData[end_date]),
        SprintmetaData[sprint] = SelectedSprint
    )
RETURN
    IF (
        NOT(ISBLANK(SelectedSprint)) &&
        'Calendar'[date] >= SprintStart &&
        'Calendar'[date] <= SprintEnd,
        1,
        0
    )
